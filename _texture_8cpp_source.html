293 gl_font_t *Texture::generateFont(ttf_texture_t *texture)
294 {
295  const float spacing = 4.0;
296  unsigned int i;
297  float u, v, u2, v2;
298  int h;
299  gl_font_t *font;
300 
301 
302  if (!texture)
303  return NULL;
304 
305  printf("Generating gl font from texture...\n");
306 
307  font = new gl_font_t;
308  font->utf8Offset = texture->utf8Offset;
309  font->count = texture->count;
310  font->textureId = loadBuffer(texture->texture,
311  texture->width, texture->width, RGBA, 32);
312 
313  glColor3f(1.0, 1.0, 1.0);
314  font->drawListBase = glGenLists(texture->count);
315  glBindTexture(GL_TEXTURE_2D, font->textureId);
316 
317  for (i = 0; i < texture->count; i++)
318  {
319  /* Generate texture coordinates for this TTF glyph */
320  u = (float)texture->glyphs[i].x / (float)texture->width;
321  v = (float)texture->glyphs[i].y / (float)texture->width;
322  u2 = (((float)texture->glyphs[i].x + (float)texture->glyphs[i].w) /
323  (float)texture->width);
324  v2 = (((float)texture->glyphs[i].y + (float)texture->glyphs[i].h) /
325  (float)texture->width);
326 
327 #ifdef DEBUG_TTF_OFFSET
328  if (i+texture->utf8Offset == 'y' || i+texture->utf8Offset == 'x')
329  {
330  printf("%c: %i %i %i\n",
331  i+texture->utf8Offset,
332  texture->fontDescent,
333  texture->glyphs[i].miny, texture->glyphs[i].maxy);
334  }
335 #endif
336 
337  if (texture->glyphs[i].h < texture->fontHeight)
338  {
339  h = texture->fontHeight - texture->glyphs[i].h;
340  }
341  else
342  {
343  h = 0;
344  }
345 
346  h += -texture->fontHeight/2-(texture->fontDescent + texture->glyphs[i].miny);
347 
348  /* Make a list for this TTF glyph, one nonuniform Quad per glyph */
349  glNewList(font->drawListBase + i, GL_COMPILE);
350 
351  glBegin(GL_QUADS);
352  glTexCoord2f(u2, v); /* Top, right */
353  glVertex3i(texture->glyphs[i].w, h, 0);
354 
355  glTexCoord2f(u, v); /* Top, left */
356  glVertex3i(0, h, 0);
357 
358  glTexCoord2f(u, v2); /* Bottom, left */
359  glVertex3i(0, h+texture->glyphs[i].h, 0);
360 
361  glTexCoord2f(u2, v2); /* Bottom, right */
362  glVertex3i(texture->glyphs[i].w, h+texture->glyphs[i].h, 0);
363  glEnd();
364 
365  /* Move To The Left Of The Character */
366  glTranslated(texture->glyphs[i].w + spacing, 0, 0);
367  glEndList();
368  }
369 
370  return font;
371 }

374 ttf_texture_t *Texture::generateFontTexture(const char *filename, int pointSize,
375  unsigned int textureWidth,
376  unsigned char color[3],
377  unsigned int utf8Offset,
378  unsigned int count,
379  char verbose)
380 {
381  unsigned int i;
382  int xx = 0, yy = 0, hh = 0, k, h, w, offset;
383  unsigned char b;
384  unsigned char *image;
385  SDL_Surface* glyph;
386  SDL_Color sdlColor;
387  TTF_Font *font;
388  ttf_texture_t *texture;
389  FILE *f;
390 
391 
392  sdlColor.r = color[0];
393  sdlColor.g = color[1];
394  sdlColor.b = color[2];
395 
396  if (!(mFlags & fUseSDL_TTF))
397  {
398  printf("SDL_TTF couldn't be used... exiting\n");
399  return NULL;
400  }
401 
402  if (pointSize < 8)
403  {
404  pointSize = 8;
405  }
406 
407  /* Work around for TTF_OpenFont for file not found segfault */
408  f = fopen(filename, "rb");
409  if (!f)
410  {
411  printf("generateFontTexture> Couldn't load '%s'\n", filename);
412  perror(filename);
413  return NULL;
414  }
415  fclose(f);
416 
417  // Open the font file at the requested point size
418  font = TTF_OpenFont(filename, pointSize);
419 
420  if (font == NULL)
421  {
422  fprintf(stderr, "generateFontTexture> Couldn't load %d pt font from %s: %s\n",
423  pointSize, filename, SDL_GetError());
424  return NULL;
425  }
426 
427  // TTF_STYLE_NORMAL, TTF_STYLE_BOLD, TTF_STYLE_ITALIC, TTF_STYLE_UNDERLINE
428  int renderStyle = TTF_STYLE_NORMAL;
429  TTF_SetFontStyle(font, renderStyle);
430 
431  /* Allocate a new TTF font texture */
432  printf("Creating font texture from '%s'...\n", filename);
433  texture = new ttf_texture_t;
434  texture->width = textureWidth;
435  texture->utf8Offset = utf8Offset;
436  texture->count = count;
437  texture->glyphs = new ttf_glyph_t[count];
438  texture->texture = new unsigned char[textureWidth*textureWidth*4];
439  memset(texture->texture, 0, textureWidth*textureWidth*4);
440 
441  texture->fontHeight = TTF_FontHeight(font);
442  texture->fontAscent = TTF_FontAscent(font);
443  texture->fontDescent = TTF_FontDescent(font);
444  texture->fontSpacing = TTF_FontLineSkip(font);
445 
446  for (i = 0; i < count; ++i)
447  {
448  glyph = TTF_RenderGlyph_Blended(font, (Uint16)(i + utf8Offset), sdlColor);
449 
450  if (glyph)
451  {
452  image = (unsigned char*)glyph->pixels;
453 
454  TTF_GlyphMetrics(font, (Uint16)(i + utf8Offset),
455  &texture->glyphs[i].minx, &texture->glyphs[i].maxx,
456  &texture->glyphs[i].miny, &texture->glyphs[i].maxy,
457  &texture->glyphs[i].advance);
458 
459  texture->glyphs[i].w = glyph->w;
460  texture->glyphs[i].h = glyph->h;
461 
462  if (xx + texture->glyphs[i].w > ((int)textureWidth - 1))
463  {
464  yy += hh;
465  hh = 2;
466  xx = 2;
467  texture->glyphs[i].x = 0;
468  texture->glyphs[i].y = yy;
469  }
470  else
471  {
472  (texture->glyphs[i].h > hh) ? hh = texture->glyphs[i].h: 0;
473  texture->glyphs[i].x = xx;
474  texture->glyphs[i].y = yy;
475  }
476 
477  xx += glyph->w;
478 
479  if (verbose)
480  {
481  printf("0x%x : %ix%i @ %i, %i\n", i + utf8Offset,
482  texture->glyphs[i].w, texture->glyphs[i].h,
483  texture->glyphs[i].x, texture->glyphs[i].y);
484  }
485 
486  /* Blit @ xx, yy - in pixels */
487  for (k = 0; k < glyph->w*glyph->h; ++k)
488  {
489  w = texture->glyphs[i].x + k%glyph->w;
490  h = texture->glyphs[i].y + k/glyph->w;
491  offset = (w + h*textureWidth);
492 
493  if (verbose)
494  {
495  printf("Offset: %i; Pixel: %i,%i; Data: 0x%08x\n",
496  offset, w, h, *((unsigned int *)(void *)&image[k*4]));
497  }
498 
499  /* 32-bit ARGB to RGBA */
500  b = image[k*4+3];
501  texture->texture[offset*4] = image[k*4] = image[k*4+1];
502  texture->texture[offset*4+1] = image[k*4+1] = image[k*4+2];
503  texture->texture[offset*4+2] = image[k*4+2] = image[k*4+3];
504  texture->texture[offset*4+3] = image[k*4+3] = b;
505  }
506  }
507  }
508 
509  TTF_CloseFont(font);
510 
511  return texture;
512 }

688 int Texture::loadBufferSlot(unsigned char *image,
689  unsigned int width, unsigned int height,
690  ColorMode mode, unsigned int bpp,
691  unsigned int slot)
692 {
693  unsigned char bytes;
694  unsigned int glcMode;
695 
696 
697  if (!mTextureIds || slot >= mTextureLimit)
698  {
699  printf("Texture::Load> ERROR Not initialized or out of free slots\n");
700  return -1000;
701  }
702 
703  if (!width || !height || !image)
704  {
705  printf("Texture::Load> ERROR Assertion 'image is valid' failed\n");
706  return -1;
707  }
708 
709  switch (mode)
710  {
711  case GREYSCALE:
712  if (bpp != 8)
713  {
714  printf("Texture::Load> ERROR Unsupported GREYSCALE, %i bpp\n", bpp);
715  return -2;
716  }
717  bytes = 1;
718  glcMode = GL_LUMINANCE;
719  break;
720  case RGB:
721  if (bpp != 24)
722  {
723  printf("Texture::Load> ERROR Unsupported RGB, %i bpp\n", bpp);
724  return -2;
725  }
726  bytes = 3;
727  glcMode = GL_RGB;
728  break;
729  case ARGB:
730  if (bpp == 32)
731  {
732  convertARGB32bppToRGBA32bpp(image, width, height);
733  }
734  else
735  {
736  printf("Texture::Load> ERROR Unsupported ARGB, %i bpp\n", bpp);
737  return -2;
738  }
739  bytes = 4;
740  glcMode = GL_RGBA;
741  break;
742  case RGBA:
743  if (bpp != 32)
744  {
745  printf("Texture::Load> ERROR Unsupported RGBA, %i bpp\n", bpp);
746  return -2;
747  }
748  bytes = 4;
749  glcMode = GL_RGBA;
750  break;
751  }
752 
753  glClearColor(0.0, 0.0, 0.0, 0.0);
754  glEnable(GL_DEPTH_TEST);
755  glShadeModel(GL_SMOOTH);
756 
757  glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
758 
759  glBindTexture(GL_TEXTURE_2D, mTextureIds[slot]);
760 
761  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
762  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
763  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
764 
765  if (mFlags & fUseMipmaps)
766  {
767  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER,
768  GL_NEAREST_MIPMAP_LINEAR);
769  //glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
770  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER,
771  GL_LINEAR_MIPMAP_LINEAR);
772

1058  delete [] timage;
1059

1062  }

1064  // Copy user data to float format.
1065  for (i = 0; i < original_height * original_width * components; ++i)
1066  {
1067  tempin[i] = (float)image[i];

1069

1083  }

1086  sy = (float)(original_height - 1);
1087  }
1088

1090  {
1091  /* Magnify both width and height: use weighted sample of 4 pixels */
1092  int i0, i1, j0, j1;
1093  float alpha, beta;
1094  float* src00;
1095  float* src01;
1096  float* src10;
1097  float* src11;
1098  float s1, s2;
1099  float* dst;
1100

1102  {
1103  i0 = (int)(i * sy);
1104  i1 = i0 + 1;
1105 
1106  if (i1 >= original_height)
1107  {
1108  i1 = original_height - 1;
1109  }
1110

1117

1119  {
1120  j1 = original_width - 1;
1121  }
1122

1124

1126  src00 = tempin + (i0 * original_width + j0) * components;
1127  src01 = tempin + (i0 * original_width + j1) * components;
1128  src10 = tempin + (i1 * original_width + j0) * components;
1129  src11 = tempin + (i1 * original_width + j1) * components;
1130

1132

1134  {
1135  s1 = *src00++ * (1.0f - beta) + *src01++ * beta;
1136  s2 = *src10++ * (1.0f - beta) + *src11++ * beta;
1137  *dst++ = s1 * (1.0f - alpha) + s2 * alpha;
1138  }
1139  }
1140  }
1141  }
1142  else
1143  {
1144  /* Shrink width and/or height: use an unweighted box filter */
1145  int i0, i1;

1147  int ii, jj;
1148  float sum;

1166  if (j1 >= original_width)
1167  {
1168  j1 = original_width - 1;
1169  }
1170

1172 
1173  /* Compute average of pixels in the rectangle (i0,j0)-(i1,j1) */
1174  for (k = 0; k < components; ++k)
1175  {
1176  sum = 0.0;
1177 
1178  for (ii = i0; ii <= i1; ++ii)
1179  {
1180  for (jj = j0; jj <= j1; ++jj)
1181  {
1182  sum += *(tempin + (ii * original_width + jj)
1183  * components + k);
1184  }
1185  }

1191  }
1192  }

1198  }
1199

1201  delete[] tempin;
